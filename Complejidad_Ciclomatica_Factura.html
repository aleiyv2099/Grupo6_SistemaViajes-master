1. Método 1: calculateTotalAmount

1.1 Código Fuente

public double calculateTotalAmount(List<Integer> codigosReserva) {
    if (codigosReserva == null || codigosReserva.isEmpty()) return 0.0;   // D1
    double total = 0.0;
    String sql = "SELECT PrecioPasaje FROM reservas WHERE CodigoReserva = ?";
    try (Connection con = dbManager.getActiveConnection();
         PreparedStatement ps = con.prepareStatement(sql)) {
        for (Integer codigo : codigosReserva) {                           // D2
            ps.setInt(1, codigo);
            try (ResultSet rs = ps.executeQuery()) {
                if (rs.next()) total += rs.getDouble("PrecioPasaje");     // D3
            }
        }
    } catch (SQLException e) {                                            // D4
        LOGGER.log(Level.SEVERE, "Error al calcular monto total", e);
    }
    return total;
}

2. Método 2: registerInvoiceWithTransaction

2.1 Código Fuente

public boolean registerInvoiceWithTransaction(FacturaDTO factura) {
    String sqlFactura  = "INSERT INTO facturas (...) VALUES (?, ?, ?, ?, ?)";
    String sqlReservas = "UPDATE reservas SET Estado = 0 WHERE CodigoCliente = ?";
    String sqlDetalle  = "INSERT INTO detalle_factura (CodigoFactura, CodigoReserva) VALUES (?, ?)";
    try (Connection con = dbManager.getActiveConnection()) {
        con.setAutoCommit(false);
        try {
            // Insertar factura y obtener llave generada
            int codigoFactura;
            try (PreparedStatement ps = con.prepareStatement(sqlFactura, RETURN_GENERATED_KEYS)) {
                ps.executeUpdate();
                try (ResultSet keys = ps.getGeneratedKeys()) {
                    if (!keys.next()) throw new SQLException("No se generó CodigoFactura"); // D1
                    codigoFactura = keys.getInt(1);
                }
            }
            // Marcar reservas como inactivas
            try (PreparedStatement ps = con.prepareStatement(sqlReservas)) { ps.executeUpdate(); }
            // Insertar detalle
            List<Integer> reservas = factura.getReservationList();
            if (reservas == null) reservas = new ArrayList<>();           // D2
            try (PreparedStatement ps = con.prepareStatement(sqlDetalle)) {
                for (Integer codigoReserva : reservas) {                   // D3
                    ps.setInt(1, codigoFactura); ps.setInt(2, codigoReserva); ps.addBatch();
                }
                ps.executeBatch();
            }
            con.commit();
            return true;
        } catch (SQLException e) {                                         // D4
            con.rollback();
            return false;
        }
    } catch (SQLException e) {                                             // D5
        LOGGER.log(Level.SEVERE, "Error de conexión", e);
        return false;
    }
}
